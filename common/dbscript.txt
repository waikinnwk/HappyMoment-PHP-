CakeEvent :

Create TABLE CakeEvent(
CE_ID VARCHAR(20) NOT NULL PRIMARY KEY,
CE_GOOGLE_EVENT_ID VARCHAR(4000),
CE_TITLE VARCHAR(200),
CE_QUOTA INT,
CE_PRICE INT,
CE_DATE DATE,
CE_TIME VARCHAR(20),
CE_LOC VARCHAR(100),
CE_STATUS VARCHAR(1)
);

ALTER TABLE CakeEvent ADD COLUMN CE_EMAIL VARCHAR(1) DEFAULT '1';
ALTER TABLE CakeEvent ADD COLUMN CE_CCG_ID int;
CakeApplication :

Create TABLE CakeApplication(
CA_ID VARCHAR(25) NOT NULL PRIMARY KEY,
CA_CE_ID VARCHAR(20) NOT NULL ,
CA_NAME VARCHAR(100),
CA_EMAIL VARCHAR(100),
CA_TEL VARCHAR(20),
CA_REMARKS VARCHAR(2000),
CA_NOP INT,
CA_WHATSAPP VARCHAR(2),
CA_STATUS VARCHAR(2),
CA_CREATE_DATE TIMESTAMP DEFAULT NOW(),
CA_PREV_ID VARCHAR(25),
CONSTRAINT FOREIGN KEY (CA_CE_ID) REFERENCES CakeEvent(CE_ID) 
);

CREATE TABLE CakeTab (
     CT_ID INT NOT NULL AUTO_INCREMENT,
     CT_NAME VARCHAR(30) NOT NULL,
	 CT_STATUS VARCHAR(1) ,
     PRIMARY KEY (CT_ID)
);

CREATE TABLE CakeClassGroup (
     CCG_ID INT NOT NULL AUTO_INCREMENT,
	 CCG_CT_ID INT NOT NULL,
     CCG_NAME VARCHAR(30) NOT NULL,
	 CCG_ALBUM_ID VARCHAR(100) NOT NULL,
	 CCG_STATUS VARCHAR(1) ,
     PRIMARY KEY (CCG_ID),
	 CONSTRAINT FOREIGN KEY (CCG_CT_ID) REFERENCES CakeTab(CT_ID) 
);
